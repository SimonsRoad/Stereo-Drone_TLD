135  boost::shared_ptr<SubState_<SubDimension> > substate;
136  if (!name.empty()) substate = getSubState<SubDimension>(name);
137  if (!substate) {

139  substates_.push_back(boost::static_pointer_cast<SubState>(substate));
140  if (!name.empty()) substates_by_name_[name] = SubStateWPtr(boost::static_pointer_cast<SubState>(substate));
141  }
142  substates_by_model_[model] = SubStateWPtr(boost::static_pointer_cast<SubState>(substate));
